1.6.1 [invariant]

  • equals(P, p2) =def (x(P) = x(p2) ∧ y(P) = y(p2) ∧ z(P) = z(p2))
  • equals(P, i, j, k) =def (x(P) = i ∧ y(P) = j ∧ z(P) = k)
  • collsion(P, p2) =def equals(P, p2) ∨ equals(P, x(p2) - 1, y(p2), z(p2))
